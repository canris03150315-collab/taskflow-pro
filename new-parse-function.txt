function parseAnnouncementJson(ann) {
  if (!ann) return ann;
  let readByArray = [];
  try {
    readByArray = ann.read_by ? JSON.parse(ann.read_by) : [];
  } catch (e) {
    readByArray = [];
  }
  return {
    id: ann.id,
    title: ann.title,
    content: ann.content,
    priority: ann.priority,
    createdBy: ann.created_by,
    createdAt: ann.created_at,
    updatedAt: ann.updated_at,
    readBy: readByArray,
    read_by: readByArray
  };
}
